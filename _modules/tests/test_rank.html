[docs]def test_pca_full():
    """ Test PCA (full rank)"""
    D = np.vstack((np.eye(3), -np.eye(3)))
    # Axis 0 is most variability
    # Axis 1 is next
    # Axis 2 is smallest
    D[[1, 4], 1] *= 1e-1
    D[[2, 5], 2] *= 1e-2

    T, W, s2 = pca(D)

    assert np.allclose(np.eye(3), np.abs(W))
    assert np.allclose(np.abs(D), np.abs(T))
